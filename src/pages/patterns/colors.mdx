import { Box, Row, Heading, Column, Text, Grid } from '@fxtrot/ui';
import { defaultColors } from 'src/lib/theme/default';

# Palette

export const styles = {
  colorItem: {
    'flexShrink': 0,
    'width': '50px',
    'height': '50px',
    'transition': '0.24s',
    'cursor': 'pointer',
    'br': '$md',
    '&:hover': {
      boxShadow: '$xs',
    },
  },
};

<Box>
  {Object.entries(defaultColors)
    .reduce((res, [name, color]) => {
      const prefix = name.split('-')[0];
      const prefixInTheMap = res.find(([title]) => title === prefix);
      if (prefixInTheMap) {
        prefixInTheMap[1][name] = color;
      } else {
        res.push([prefix, { [name]: color }]);
      }
      return res;
    }, [])
    .map(([title, colors]) => (
      <React.Fragment key={title}>
        <Heading level="3" as="h2">
          {title}
        </Heading>
        <Row wrap="wrap" gap="8">
          {Object.entries(colors).map(([name, color]) => (
            <Column gap="3" css={{ width: '140px' }} key={name}>
              <Box
                css={styles.colorItem}
                bc={`$${name}`}
                boxShadow={`$xs, 0 4px 6px -1px $colors$${name}`}
                title={name}
              />
              <Column gap="2">
                <Text size="sm" lineClamp="1" title={name}>
                  ${name}
                </Text>
                <Text size="sm" font="mono" tone="light" lineClamp="1" title={color}>
                  {color}
                </Text>
              </Column>
            </Column>
          ))}
        </Row>
      </React.Fragment>
    ))}
</Box>
