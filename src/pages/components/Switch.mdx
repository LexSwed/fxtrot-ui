## Switch

```tsx preview
<Row gap="xl">
  <Switch label="Default" />
  <Switch label="Disabled" disabled />
  <Switch label="Checked disabled" disabled defaultChecked />
</Row>
```

## Secondary text

```tsx preview
<Switch label="Format on save" secondaryLabel="(required)" required />
```

## Theming

Use `ThemeProvider` component to change the primary color of the button.

```tsx preview
<Column gap="xl">
  <Row gap="md">
    <ThemeProvider theme="rose">
      <Switch label="Format on save" gap="md" />
      <Switch label="Format on save (disabled)" disabled gap="md" />
    </ThemeProvider>
  </Row>
  <Row gap="md">
    <ThemeProvider theme="green">
      <Switch label="Format on save" />
      <Switch label="Format on save (disabled)" disabled />
    </ThemeProvider>
  </Row>
</Column>
```
