import { DocumentAddIcon } from '@heroicons/react/outline';

```tsx preview
<Button flow="column" cross="start" gap="0">
  <Row cross="center" gap="1">
    <Icon as={DocumentAddIcon} size="lg" />
    <Text textStyle="label-lg" weight={500}>
      Create
    </Text>
  </Row>
  <Row>
    <Text textStyle="label-sm">Create a new document with the default template</Text>
  </Row>
</Button>
```

## Variant

Do not use `variant="link"` with actual links, use [`LinkButton`](/components/LinkButton) component instead.

```tsx preview
<Grid gap="lg" cross="center" main="center" columns="repeat(2, auto)" css={{ p: '$4' }}>
  <Button variant="primary">Primary</Button>
  <Button disabled variant="primary">
    Primary (disabled)
  </Button>
  <Button variant="outline">Outline</Button>
  <Button variant="outline" disabled>
    Outline (disabled)
  </Button>
  <Button variant="tonal">Tonal</Button>
  <Button variant="tonal" disabled>
    Tonal (disabled)
  </Button>
  <Button variant="flat">Flat</Button>
  <Button variant="flat" disabled>
    Flat (disabled)
  </Button>
  <Button variant="link">Link</Button>
  <Button variant="link" disabled>
    Link (disabled)
  </Button>
</Grid>
```

## Theming

Use `ThemeProvider` component to change the primary color of the button.

```tsx preview
() => {
  const theme = {
    colors: {
      primary: 'hsl(228 100% 85.5%)',
      onPrimary: 'hsl(223 100% 26.1%)',
      primaryContainer: 'hsl(221 100% 35.5%)',
      onPrimaryContainer: 'hsl(0 0% 0%)',
      secondary: 'hsl(203 100% 74.9%)',
      onSecondary: 'hsl(165 100% 11%)',
      secondaryContainer: 'hsl(167 100% 15.9%)',
      onSecondaryContainer: 'hsl(162 89.9% 72.9%)',
      tertiary: 'hsl(191 100% 58.6%)',
      onTertiary: 'hsl(192 100% 12.9%)',
      tertiaryContainer: 'hsl(191 100% 18.8%)',
      onTertiaryContainer: 'hsl(195 100% 84.7%)',
      success: 'hsl(163 63.5% 61.4%)',
      onSuccess: 'hsl(165 100% 11%)',
      successContainer: 'hsl(167 100% 15.9%)',
      onSuccessContainer: 'hsl(162 89.9% 72.9%)',
      error: 'hsl(348 100% 84.9%)',
      onError: 'hsl(338 100% 20%)',
      errorContainer: 'hsl(337 100% 28.2%)',
      onErrorContainer: 'hsl(351 100% 92.5%)',
      outline: 'hsl(264deg 5% 58%)',
      background: 'hsl(204 9.1% 10.8%)',
      onBackground: 'hsl(210 7.1% 89%)',
      surface: 'hsl(204 10% 25%)',
      onSurface: 'hsl(210 7.1% 89%)',
      surfaceVariant: 'hsl(264 7% 56%)',
      onSurfaceVariant: 'hsl(270 11.3% 79.2%)',
      inverseSurface: 'hsl(210 7.1% 89%)',
      inverseOnSurface: 'hsl(204 9.1% 10.8%)',
      disabled: 'hsl(0 0% 89% / 8%)',
      onDisabled: 'hsl(210 5% 75%)',
    },
  };
  return (
    <ThemeProvider theme={theme} css={{ colorScheme: 'dark' }}>
      <Grid gap="lg" cross="center" main="center" columns="repeat(2, auto)" css={{ p: '$4', bc: '$background' }}>
        <Button variant="primary" gap="2">
          Outline
          <Icon as={DotsVerticalIcon} />
        </Button>
        <Button disabled variant="primary">
          Primary (disabled)
        </Button>
        <Button variant="outline">Outline</Button>
        <Button variant="outline" disabled>
          Outline (disabled)
        </Button>
        <Button variant="tonal">Tonal</Button>
        <Button variant="tonal" disabled>
          Tonal (disabled)
        </Button>
        <Button variant="flat">Flat</Button>
        <Button variant="flat" disabled>
          Flat (disabled)
        </Button>
        <Button variant="link">Link</Button>
        <Button variant="link" disabled>
          Link (disabled)
        </Button>
      </Grid>
    </ThemeProvider>
  );
};
```

## With elements

If you only want to put [`Icon`](/components/Icon) inside the `Button`, use [`IconButton`](/components/IconButton) component instead.

```tsx preview
<Column cross="start" gap="xl" css={{ p: '$8' }}>
  <Button variant="primary">
    <Icon as={DotsVerticalIcon} />
    More
  </Button>
</Column>
```
