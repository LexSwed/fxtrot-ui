import { Playground } from 'dokz';
import { Box, Button, Menu, Icon, Text, ThemeProvider, Dialog, Picker, Item } from '@fxtrot/ui';
import { ChevronDownIcon } from '@heroicons/react/solid';
import { TrashIcon } from '@heroicons/react/outline';

## Menu

For big lists use `value` prop on `Menu.Item` and `onAction` callback to handle click on the Menu.Item with `action`.

<Playground>
  <Box p="$8">
    <Menu>
      <Button>
        <span>Open menu</span>
        <Icon as={ChevronDownIcon} size="lg" />
      </Button>
      <Menu.List>
        <Menu.Item onSelect={() => window.alert('copy')}>
          <Icon as="svg" />
          Copy
        </Menu.Item>
        <Menu.Item onSelect={() => window.alert('cut')}>
          <Icon as="svg" />
          Cut
        </Menu.Item>
        <Menu.Separator />
        <Menu.Item onSelect={() => window.alert('delete')} css={{ color: '$danger' }}>
          <Icon as={TrashIcon} color="$danger" />
          Delete
        </Menu.Item>
      </Menu.List>
    </Menu>
  </Box>
</Playground>

You can modify `offset` from the `side` and along the `align`ment.

<Playground>
  <ThemeProvider theme="orange">
    <Box p="$8">
      <Menu>
        <Button>
          <span>Open menu</span>
          <Icon as={ChevronDownIcon} size="lg" />
        </Button>
        <Menu.List sideOffset={12} side="right" align="start">
          {Array(20)
            .fill(null)
            .map((el, i) => (
              <Menu.Item key={i} onSelect={() => window.alert(i)}>
                Item {i}
              </Menu.Item>
            ))}
        </Menu.List>
      </Menu>
    </Box>
  </ThemeProvider>
</Playground>

## Issues

- `ArrowUp` and `ArrowDown` just open the menu, first option is always focused
- No <kbd>Home</kbd>, <kbd>End</kbd>, <kbd>Page Up</kbd> or <kbd>Page Down</kbd> support
- No typeahead support

<Playground>
  <Box p="$8">
    <Menu>
      <Button variant="outline">
        <Icon as={ChevronDownIcon} size="lg" />

      <Menu.List align="start" side="bottom">

          {(close) => (
            <Dialog.Modal>
              <Dialog.Title level={3}>Delete topic?</Dialog.Title>
              <Picker>

                <Item label="Item 2" />
              </Picker>
            </Dialog.Modal>
          )}
        </Dialog>
      </Menu.List>
    </Menu>
  </Box>
</Playground>
