## MenuList

```tsx preview
() => {
  const [selected, setSelected] = React.useState('Appearance');
  return (
    <Box width="$56">
      <MenuList role="navigation">
        <Section title="Settings">
          {[
            {
              icon: UsersIcon,
              label: 'Profiles',
            },
            {
              icon: AdjustmentsIcon,
              label: 'Appearance',
            },
            {
              icon: GlobeIcon,
              label: 'Language',
            },
          ].map(({ icon, label }) => (
            <MenuList.Item key={label} aria-selected={selected === label} onClick={() => setSelected(label)}>
              <Icon as={icon} />
              {label}
            </MenuList.Item>
          ))}
        </Section>
      </MenuList>
    </Box>
  );
};
```
