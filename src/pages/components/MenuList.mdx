## MenuList

```tsx preview
() => {
  const [selected, setSelected] = React.useState('Appearance');
  return (
    <div className="w-[240px]">
      <MenuList role="navigation">
        <Section title="Settings">
          {[
            {
              icon: UsersIcon,
              label: 'Profiles',
            },
            {
              icon: AdjustmentsHorizontalIcon,
              label: 'Appearance',
            },
            {
              icon: GlobeEuropeAfricaIcon,
              label: 'Language',
            },
          ].map(({ icon, label }) => (
            <MenuList.Item key={label} aria-selected={selected === label} onClick={() => setSelected(label)}>
              <Icon as={icon} />
              {label}
            </MenuList.Item>
          ))}
        </Section>
      </MenuList>
    </div>
  );
};
```
