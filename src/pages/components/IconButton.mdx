# IconButton

Same as [`Button`](/components/Button) component, but when used only with an icon inside. Provides prop check for `ARIA` label and applies some light styling. All the existing props of the [`Button`](/components/Button) are supported as well.

## Example

```tsx preview
<Column gap="lg">
  <Row gap="md" css={{ px: '$10' }}>
    <IconButton label="Move right" size="xs" variant="primary">
      <Icon as={ArrowNarrowRightIcon} size="sm" />
    </IconButton>
    <IconButton label="Move right" size="xs" variant="outline">
      <Icon as={ArrowNarrowRightIcon} size="sm" />

    <IconButton label="Move right" size="xs" variant="neutral">
```

## Theming

Use `ThemeProvider` component to change the primary color of the button.

```tsx preview
() => {
  const theme = {
    colors: {
      primary: 'hsl(228 100% 85.5%)',
      onPrimary: 'hsl(223 100% 26.1%)',
      primaryContainer: 'hsl(221 100% 35.5%)',
      onPrimaryContainer: 'hsl(0 0% 0%)',
      secondary: 'hsl(203 100% 74.9%)',
      onSecondary: 'hsl(165 100% 11%)',
      secondaryContainer: 'hsl(167 100% 15.9%)',
      onSecondaryContainer: 'hsl(162 89.9% 72.9%)',
      tertiary: 'hsl(191 100% 58.6%)',
      onTertiary: 'hsl(192 100% 12.9%)',
      tertiaryContainer: 'hsl(191 100% 18.8%)',
      onTertiaryContainer: 'hsl(195 100% 84.7%)',
      success: 'hsl(163 63.5% 61.4%)',
      onSuccess: 'hsl(165 100% 11%)',
      successContainer: 'hsl(167 100% 15.9%)',
      onSuccessContainer: 'hsl(162 89.9% 72.9%)',
      error: 'hsl(348 100% 84.9%)',
      onError: 'hsl(338 100% 20%)',
      errorContainer: 'hsl(337 100% 28.2%)',
      onErrorContainer: 'hsl(351 100% 92.5%)',
      outline: 'hsl(264deg 5% 58%)',
      background: 'hsl(204 9.1% 10.8%)',
      onBackground: 'hsl(210 7.1% 89%)',
      surface: 'hsl(204 10% 25%)',
      onSurface: 'hsl(210 7.1% 89%)',
      surfaceVariant: 'hsl(264 7% 56%)',
      onSurfaceVariant: 'hsl(270 11.3% 79.2%)',
      inverseSurface: 'hsl(210 7.1% 89%)',
      inverseOnSurface: 'hsl(204 9.1% 10.8%)',
      disabled: 'hsl(0 0% 89% / 8%)',
      onDisabled: 'hsl(210 5% 75%)',
    },
  };
  return (
    <ThemeProvider theme={theme}>
      <Column gap="xl" css={{ bc: '$surface', p: '$4' }}>
        <Row gap="md" css={{ px: '$10' }}>
          <IconButton label="Move right" variant="primary">
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
          <IconButton label="Move right" disabled variant="primary">
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
        </Row>
        <Row gap="md" css={{ px: '$10' }}>
          <IconButton label="Move right" variant="neutral">
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
          <IconButton label="Move right" variant="neutral" disabled>
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
        </Row>
        <Row gap="md" css={{ px: '$10' }}>
          <IconButton label="Move right" variant="flat">
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
          <IconButton label="Move right" variant="flat" disabled>
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
        </Row>
        <Row gap="md" css={{ px: '$10' }}>
          <IconButton label="Move right" variant="outline">
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
          <IconButton label="Move right" variant="outline" disabled>
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
        </Row>
        <Row gap="md" css={{ px: '$10' }}>
          <IconButton label="Move right" variant="link">
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
          <IconButton label="Move right" variant="link" disabled>
            <Icon as={ArrowNarrowRightIcon} />
          </IconButton>
        </Row>
      </Column>
    </ThemeProvider>
  );
};
```
