## TextArea

`TextField`-like element but `textarea` and with automatic size adjustment.

```tsx preview
<Row gap="md">
  <TextArea label="Enter your name" placeholder="John Doe" />
  <TextArea label="Enter your name" placeholder="John Doe" secondaryLabel="(disabled)" disabled />
  <TextArea label="Enter your name" placeholder="John Doe" secondaryLabel="(read only)" readOnly />
</Row>
```

## Validity

```tsx preview
() => {
  const [value, setValue] = React.useState('me@email.com');
  const isValid = React.useMemo(() => /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/.test(value), [value]);
  return (
    <Row gap="md" cross="end">
      <TextArea
        cross="stretch"
        validity={isValid ? 'valid' : 'invalid'}
        label="Email"
        hint={!isValid ? 'Check that your email is typed correctly' : ' '}
        placeholder="john.doe@mail.com"
        value={value}
        onChange={setValue}
      />
      <TextArea
        cross="stretch"
        validity={isValid ? 'valid' : 'invalid'}
        label="Email"
        hint={!isValid ? 'Check that your email is typed correctly' : ' '}
        placeholder="john.doe@mail.com"
        value={value}
        onChange={setValue}
        variant="underlined"
      />
      <TextArea
        cross="stretch"
        validity={isValid ? 'valid' : 'invalid'}
        hint={!isValid ? 'Check that your email is typed correctly' : ' '}
        placeholder="john.doe@mail.com"
        value={value}
        onChange={setValue}
        variant="flat"
      />
    </Row>
  );
};
```
