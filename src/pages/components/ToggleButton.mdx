## Stateful

<Playground>
  {() => {
    const [bold, setBold] = React.useState(false);
    return (
      <ToggleButton
        pressed={bold}
        onPressedChange={setBold}
        size="sm"
        aria-label={bold ? 'Turn off bold' : 'Turn on bold'}
      >
        <Icon as={BsTypeBold} />
      </ToggleButton>
    );
  }}
