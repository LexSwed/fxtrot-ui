## Variant

```tsx preview
<Grid gap="lg" cross="center" main="center" columns="2" placeItems="start" className="p-4">
  <LinkButton href="#" variant="primary">
    Primary
  </LinkButton>
  <LinkButton href="#" disabled variant="primary">
    Primary (disabled)
  </LinkButton>
  <LinkButton href="#" variant="outline">
    Outline
  </LinkButton>
  <LinkButton href="#" variant="outline" disabled>
    Outline (disabled)
  </LinkButton>
  <LinkButton href="#" variant="tonal">
    Secondary (default)
  </LinkButton>
  <LinkButton href="#" variant="tonal" disabled>
    Secondary (disabled)
  </LinkButton>
  <LinkButton href="#" variant="flat">
    Flat
  </LinkButton>
  <LinkButton href="#" variant="flat" disabled>
    Flat (disabled)
  </LinkButton>
  <LinkButton href="#" variant="link">
    Link
  </LinkButton>
  <LinkButton href="#" variant="link" disabled>
    Link (disabled)
  </LinkButton>
</Grid>
```
