---
title: LinkButton
---

Link element that looks like a button.

## Variant

```tsx preview
<Grid gap="lg" cross="center" main="center" columns="repeat(2, auto)">
  <LinkButton href="#" variant="primary">
    Primary
  </LinkButton>
  <LinkButton href="#" disabled variant="primary">
    Primary (disabled)
  </LinkButton>
  <LinkButton href="#" variant="outline">
    Outline
  </LinkButton>
  <LinkButton href="#" variant="outline" disabled>
    Outline (disabled)
  </LinkButton>
  <LinkButton href="#" variant="secondary">
    Secondary (default)
  </LinkButton>
  <LinkButton href="#" variant="secondary" disabled>
    Secondary (disabled)
  </LinkButton>
  <LinkButton href="#" variant="flat">
    Flat
  </LinkButton>
  <LinkButton href="#" variant="flat" disabled>
    Flat (disabled)
  </LinkButton>
  <LinkButton href="#" variant="link">
    Link
  </LinkButton>
  <LinkButton href="#" variant="link" disabled>
    Link (disabled)
  </LinkButton>
</Grid>
```

## Theming

Use `ThemeProvider` component to change the primary color of the LinkButton.

```tsx preview
<ThemeProvider theme="purple">
  <Column gap="xl">
    <Row gap="sm" css={{ px: '$10' }}>
      <LinkButton href="#" variant="primary">
        Primary
      </LinkButton>
      <LinkButton href="#" disabled variant="primary">
        Primary (disabled)
      </LinkButton>
    </Row>
    <Row gap="sm" css={{ px: '$10' }}>
      <LinkButton href="#" variant="secondary">
        Secondary
      </LinkButton>
      <LinkButton href="#" variant="secondary" disabled>
        Secondary (disabled)
      </LinkButton>
    </Row>
    <Row gap="sm" css={{ px: '$10' }}>
      <LinkButton href="#" variant="flat">

  </Column>
</ThemeProvider>
```
