# ActionGroup

<Playground>
  {() => {
    const [fontStyle, setFontStyle] = React.useState(null);

          <Icon as={BsTypeItalic} />
        </ToggleButton>
        <ToggleButton value="underline">

      </ActionGroup>
    );
  }}
</Playground>
