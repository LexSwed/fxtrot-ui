# ActionGroup

```tsx preview
() => {
  const [fontStyle, setFontStyle] = React.useState(null);
  return (
    <ActionGroup value={fontStyle} onValueChange={setFontStyle}>
      <ToggleButton icon={BsTypeBold} label="bold" value="bold" />
      <ToggleButton icon={BsTypeItalic} label="italic" value="italic" />
      <ToggleButton icon={BsTypeUnderline} label="underline" value="underline" />
    </ActionGroup>
  );
};
```

## Multiple selection

```tsx preview
() => {
  const [fontStyle, setFontStyle] = React.useState([]);
  return (
    <ActionGroup value={fontStyle} onValueChange={setFontStyle} type="multiple">
      <ToggleButton icon={BsTypeBold} label="bold" value="bold" size="sm" />
      <ToggleButton icon={BsTypeItalic} label="italic" value="italic" size="sm" />
      <ToggleButton icon={BsTypeUnderline} label="underline" value="underline" size="sm" />
    </ActionGroup>
  );
};
```
