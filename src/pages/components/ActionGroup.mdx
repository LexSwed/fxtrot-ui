# ActionGroup

```tsx preview
() => {
  const [fontStyle, setFontStyle] = React.useState(null);
  return (
    <ActionGroup value={fontStyle} onValueChange={setFontStyle}>
      <ToggleButton value="bold">
        <Icon as={BsTypeBold} />
      </ToggleButton>
      <ToggleButton value="italic">
        <Icon as={BsTypeItalic} />
      </ToggleButton>
      <ToggleButton value="underline">
        <Icon as={BsTypeUnderline} />
      </ToggleButton>
    </ActionGroup>
  );
};
```

## Multiple selection

```tsx preview
() => {
  const [fontStyle, setFontStyle] = React.useState([]);
  return (
    <ActionGroup value={fontStyle} onValueChange={setFontStyle} type="multiple">
      <ToggleButton value="bold">
        <Icon as={BsTypeBold} />
      </ToggleButton>
      <ToggleButton value="italic">
        <Icon as={BsTypeItalic} />
      </ToggleButton>
      <ToggleButton value="underline">
        <Icon as={BsTypeUnderline} />
      </ToggleButton>
    </ActionGroup>
  );
};
```
