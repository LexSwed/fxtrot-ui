## Validity

```tsx preview
() => {
  const [value, setValue] = React.useState('me@email.com');
  const isValid = React.useMemo(() => /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$/.test(value), [value]);
  return (
    <Row gap="md" cross="end">
      <TextField
        cross="stretch"
        validity={isValid ? 'valid' : 'invalid'}
        label="Email"
        hint={!isValid ? 'Check that your email is typed correctly' : ' '}
        placeholder="john.doe@mail.com"
        value={value}
        onChange={setValue}
      />
      <TextField
        cross="stretch"
        validity={isValid ? 'valid' : 'invalid'}
        label="Email"
        hint={!isValid ? 'Check that your email is typed correctly' : ' '}
        placeholder="john.doe@mail.com"
        value={value}
        onChange={setValue}
        variant="underlined"
      />
      <TextField
        cross="stretch"
        validity={isValid ? 'valid' : 'invalid'}
        hint={!isValid ? 'Check that your email is typed correctly' : ' '}
        placeholder="john.doe@mail.com"
        value={value}
        onChange={setValue}
        variant="flat"
      />
    </Row>
  );
};
```
